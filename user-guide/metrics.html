Examples

obs = np.array([0.3, 2.1, -1.0]) mod = np.array([0.0, 2.3, 1.0]) bias(obs, mod) np.float64(0.6333333333333332) max_error(obs, mod) np.float64(2.0) rmse(obs, mod) np.float64(1.173314393786536) urmse(obs, mod) np.float64(0.9877021593352702) mae(obs, mod) np.float64(0.8333333333333331) mape(obs, mod) np.float64(103.17460317460316) nse(obs, mod) np.float64(0.14786795048143053) r2(obs, mod) np.float64(0.14786795048143053) mef(obs, mod) np.float64(0.9231099877688299) si(obs, mod) np.float64(0.8715019052958266) spearmanr(obs, mod) np.float64(0.5) willmott(obs, mod) np.float64(0.7484604452865941) hit_ratio(obs, mod, a=0.5) np.float64(0.6666666666666666) ev(obs, mod) np.float64(0.39614855570839064)
